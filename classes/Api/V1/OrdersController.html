def create
  @order = Order.new(order_params.reverse_merge(shop_id: params['shop_id']))

  if @order.save
    render json: @order, status: :created, location: api_v1_shop_orders_url(@order)
  else
    render json: @order.errors, status: :unprocessable_entity
  end
end
